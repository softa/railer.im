initial_scores Structure
F-Key Name Type Description
id integer
initial_score double precision 
SELECT u.id
, (
     (
           (
            SELECT COALESCE
                 (sum
                       (
                             (
                                   (
                                         (r.watchers - 1)
                                   )::double precision + COALESCE
                                   (r.score
                                         , (0)::double precision
                                   )
                             )
                       )
                       , (0)::double precision
                 ) AS "coalesce"
              FROM repositories r 
             WHERE (r.user_id = u.id)
           ) + 
           (
                 (
                       (
                        SELECT count
                             (*) AS count 
                          FROM git_followers g 
                         WHERE (g.followee_id = u.id)
                       ) * 5
                 )
           )::double precision
     ) + 
     (
           (
            SELECT COALESCE
                 (sum
                       (r.downloads)
                       , (0)::bigint
                 ) AS "coalesce"
              FROM (rubygems r 
                    JOIN authorships a 
                      ON (
                             (a.rubygem_id = r.id)
                       )
                 )
             WHERE (a.user_id = u.id)
           )
     )::double precision
) AS initial_score 
FROM users u;
